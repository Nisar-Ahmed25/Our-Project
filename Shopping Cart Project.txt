import java.util.HashMap;
import java.util.Map;
import java.util.Scanner;

 class ShoppingCart {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        // Student 1: Create a map to store items and their prices
        Map<String, Double> items = new HashMap<>();

        System.out.println("Welcome to the Shopping Cart App!");
        System.out.println("Let's start by creating your shopping account.");

        System.out.print("Enter your name: ");
        String userName = scanner.nextLine();

        // Student 2: Main loop for adding items
        boolean addingItems = true;
        while (addingItems) {
            System.out.print("Enter the name of the item (or type 'done' to finish): ");
            String itemName = scanner.nextLine();

            if (itemName.equalsIgnoreCase("done")) {
                addingItems = false;
            } else {
                System.out.print("Enter the price of the item: ");
                double itemPrice = scanner.nextDouble();
                scanner.nextLine(); // Student 1: Consume newline character
                
                // Student 2: Add item and price to the map
                items.put(itemName, itemPrice);
            }
        }

        // Student 1: Displaying the items and their prices
        System.out.println("\nItems in your shopping cart:");
        for (Map.Entry<String, Double> entry : items.entrySet()) {
            System.out.println(entry.getKey() + " - $" + entry.getValue());
        }

        // Student 2: Calculating the total price
        double totalPrice = 0.0;
        for (double price : items.values()) {
            totalPrice += price;
        }

        // Student 1: Displaying the total price
        System.out.println("\nTotal price: $" + totalPrice);
        System.out.println("Thank you, " + userName + ", for shopping with us!");
    }
}
